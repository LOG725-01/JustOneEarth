Only on Plains"; cost.Add(RessourceTypes.Trees, 5); cost.Add(RessourceTypes.Minerals, 5); effectList.Add(new GainRessourceOfType(RessourceTypes.Sun, 4)); effectList.Add(new GainPointEffect(1)); targetType = CardTargetType.NeutralTileOnly; conditionList.Add(new TileMustBeOfType(TileType.Plains)); GameObject prefab = Resources.Load<GameObject>("Structures/solarFarm"); effectList.Add(new PlaceStructureEffect(prefab, new Vector3(0, 0, 0))); } }
