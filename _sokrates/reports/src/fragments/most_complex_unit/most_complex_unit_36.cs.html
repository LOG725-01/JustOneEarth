    private bool IsInsideMap(int q, int r)
    {
        bool inside = Math.Abs(q) <= radius && Math.Abs(r) <= radius && Math.Abs(-q - r) <= radius;

        if (debug) Debug.Log($"[Board] Vérification IsInsideMap({q}, {r}) → {(inside ? "À l’intérieur" : "Hors limites")}");

        return inside;
    }
