    void Update()
    {
        if (Input.GetKeyDown(KeyCode.T) && !isOpened)
        {
            Open();
            chatInput.Select();
            chatInput.ActivateInputField();
        }
        if (!isOpened && EventSystem.current.currentSelectedGameObject == chatInput.gameObject)
        {
            EventSystem.current.SetSelectedGameObject(null);
        }
    }
